----
auto utf8Decode(MemoryView view) -> Generator<CodePoint> {
  // nest++
  auto take = [&]() -> CodePoint { return *view.begin++; };
  // nest--
  // nest++
  while (!view.empty()) {
    // nest++
    auto c0 = take();
    // nest--
    // nest++
    if ((c0 & 0x80) != 0x80)
    // nest--
      // nest++
      co_yield c0;
      // nest--
    // nest++
    else if ((c0 & 0xE0) == 0xC0) { 
      // nest++
      if (view.empty()) co_return;
      // nest--
      // nest++
      auto c1 = take();
      // nest--
      // nest++
      if ((c1 & 0xC0) != 0x80) co_yield {};
      // nest--
      // nest++
      else co_yield { ((c0 & 0x1Fu) << 6) | (c1 & 0x3Fu) };
      // nest--
    }
    // nest--
    // nest++
    else { /* … */ }
    // nest--
  }
  // nest--
}

auto tokenize(Generator<CodePoint> &input) -> Generator<Token> {
  // nest++
  auto position = TextPosition{};
  // nest--
  // nest++
  auto next = [&] { return (++input) ? *input : CodePoint{}; };
  // nest--
  // nest++
  auto cp = next();
  // nest--
  // nest++
  while (cp) {
    // nest++
    auto token = [begin = position, end = &position](auto data) {
      // nest++
      return Token{TextRange{begin, end}, data};
      // nest--
    };
    // nest--
    // nest++
    if (cp == ':') { 
      // nest++
      cp = next(); position.column++; 
      // nest--
      // nest++
      co_yield token(Colon{});
      // nest--
      // nest++
      continue;
      // nest--
    }
    // nest--
    // nest++
    // …

dynSum(unsigned long):            ; rdi = n
        add     rdi, 1            ; rdi += 1
        sete    cl                ; cl = rdi == 0
        je      .LBB1_1           ; if (rdi == 0) goto LBB1_1
        xor     eax, eax          ; eax = 0
        xor     esi, esi          ; esi = 0
.LBB1_4:
        mov     edx, esi          ; edx = esi
        add     eax, esi          ; eax += esi
        mov     esi, 0            ; esi = 0
        test    cl, 1             ; cl & 1 <=> 0
        mov     ecx, 0            ; ecx = 0
        jne     .LBB1_4           ; if (cl != 0) goto LBB1_4
        add     edx, 1            ; edx += 1
        xor     ecx, ecx          ; ecx = 0
        mov     esi, edx          ; esi = edx
        cmp     rdi, rdx          ; rdi <=> rdx
        ja      .LBB1_4           ; if (rdi > rdx) goto LBB1_4
        ret                       ; return eax
.LBB1_1:
        xor     eax, eax          ; eax = 0
        ret                       ; return eax
